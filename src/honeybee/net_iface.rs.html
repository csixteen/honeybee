1//! Utilities to manipulate network interfaces using [`netlink`].
2//!
3//! This module uses [`neli`] and [`neli-wifi`] to communicate with the Kernel via
4//! [`netlink sockets`].
5//!
6//! [`netlink`]: https://www.ietf.org/rfc/rfc3549.txt
7//! [`neli`]: https://docs.rs/neli/latest/neli/index.html
8//! [`neli-wifi`]: https://github.com/MaxVerevkin/neli-wifi/
9//! [`netlink sockets`]: https://kernel.org/doc/html/next/userspace-api/netlink/intro.html
10//!

28#[derive(Clone, Debug, SmartDefault, PartialEq)]
29pub struct NetworkInterface {
30    index: i32,
31    name: String,
32    pub ipv4: Option<Ipv4Addr>,
33    pub ipv6: Option<Ipv6Addr>,
34    wifi_info: Option<WirelessInfo>,
35    pub stats: Option<RtnlLinkStats>,
36}
37
38impl NetworkInterface {
39    pub async fn new(name: &str) -> Result<Option<Self>> {
40        let if_index = if_nametoindex(name)
41            .or_error(|| format!("Couldn't find index for interface {name}"))?
42            as i32;
43
44        let mut socket = NlSocket::new(
45            NlSocketHandle::connect(NlFamily::Route, None, &[]).error("Netlink socket error")?,
46        )
47        .error("Netlink socket error")?;
48
49        let ipv4 = get_ipv4(&mut socket, if_index).await?;
50        let ipv6 = get_ipv6(&mut socket, if_index).await?;
51        let wifi_info = WirelessInfo::new(if_index).await?;
52        let stats = RtnlLinkStats::new(&mut socket, if_index)
53            .await
54            .map_err(BoxedError)
55            .error("Couldn't get interface stats")?;
56        if stats.is_none() {
57            return Ok(None);
58        }
59
60        Ok(Some(Self {
61            index: if_index,
62            name: name.to_owned(),
63            ipv4,
64            ipv6,
65            wifi_info,
66            stats,
67        }))
68    }
69
70    pub fn ssid(&self) -> Option<String> {
71        self.wifi_info.as_ref()?.ssid.clone()
72    }
73
74    pub fn signal(&self) -> Option<f64> {
75        self.wifi_info.as_ref()?.signal
76    }
77
78    pub fn bitrate(&self) -> Option<f64> {
79        self.wifi_info.as_ref()?.bitrate
80    }
81
82    pub fn frequency(&self) -> Option<f64> {
83        self.wifi_info.as_ref()?.frequency
84    }
85}

87macro_rules! rtnetlink_recv {
88    ($socket:ident, $nl_payload:ident, $nl_type:expr, $if_index:expr, $if_attr:ident, $payload:ident : $ptype:ty => $($body:tt)*) => {
89        let nl_header = Nlmsghdr::new(
90            None,
91            $nl_type,
92            NlmFFlags::new(&[NlmF::Request, NlmF::Dump]),
93            None,
94            None,
95            NlPayload::Payload($nl_payload),
96        );
97        $socket.send(&nl_header).await?;
98
99        // Without this loop, I'd get an error `UnexpectedEOB`.
100        'msgs: loop {
101            let msgs = $socket.recv::<u16, $ptype>(&mut Vec::new()).await?;
102            for msg in msgs {
103                if msg.nl_type == u16::from(Nlmsg::Done) {
104                    break 'msgs;
105                }
106
107                if let NlPayload::Payload($payload) = msg.nl_payload {
108                    if $payload.$if_attr != $if_index {
109                        continue;
110                    }
111
112                    $($body)*
113                }
114            }
115        }
116    };
117}

119async fn get_ip_addr<const T: usize>(
120    socket: &mut NlSocket,
121    if_index: i32,
122    ifa_family: RtAddrFamily,
123) -> Result<Option<[u8; T]>, Box<dyn StdError + Send + Sync + 'static>> {
124    let ifaddrmsg = Ifaddrmsg {
125        ifa_family,
126        ifa_prefixlen: 0,
127        ifa_flags: IfaFFlags::empty(),
128        ifa_scope: 0,
129        ifa_index: 0,
130        rtattrs: RtBuffer::new(),
131    };
132
133    let mut res = None;
134
135    rtnetlink_recv!(socket, ifaddrmsg, Rtm::Getaddr, if_index, ifa_index, p: Ifaddrmsg => {
136        if RtScope::from(p.ifa_scope) != RtScope::Universe {
137            continue;
138        }
139
140        let rtattrs = p.rtattrs.get_attr_handle();
141        let Some(attr) = rtattrs
142            .get_attribute(Ifa::Local)
143            .or_else(|| rtattrs.get_attribute(Ifa::Address)) else { continue; };
144
145        if let Ok(a) = attr.rta_payload.as_ref().try_into() {
146            res = Some(a);
147        }
148    });
149
150    Ok(res)
151}
152
153async fn get_ipv4(socket: &mut NlSocket, if_index: i32) -> Result<Option<Ipv4Addr>> {
154    Ok(get_ip_addr(socket, if_index, RtAddrFamily::Inet)
155        .await
156        .map_err(BoxedError)
157        .error("Couldn't get IPv4 address")?
158        .map(Ipv4Addr::from))
159}
160
161async fn get_ipv6(socket: &mut NlSocket, if_index: i32) -> Result<Option<Ipv6Addr>> {
162    Ok(get_ip_addr(socket, if_index, RtAddrFamily::Inet6)
163        .await
164        .map_err(BoxedError)
165        .error("Couldn't get IPv6 address")?
166        .map(Ipv6Addr::from))
167}

169#[derive(Clone, Debug, SmartDefault, PartialEq)]
170pub struct WirelessInfo {
171    ssid: Option<String>,
172    signal: Option<f64>,
173    bitrate: Option<f64>,
174    frequency: Option<f64>,
175}
176
177impl WirelessInfo {
178    // https://github.com/i3/i3status/blob/main/src/print_wireless_info.c#L140
179    fn find_ssid(mut ies: &[u8]) -> Option<String> {
180        while ies.len() > 2 && ies[0] != 0 {
181            ies = &ies[(ies[1] as usize + 2)..];
182        }
183
184        if ies.len() < 2 || ies.len() < (ies[1] as usize) + 2 {
185            return None;
186        }
187
188        let ssid = &ies[2..][..(ies[1] as usize)];
189        Some(String::from_utf8_lossy(ssid).into_owned())
190    }
191
192    // https://github.com/i3/i3status/blob/main/src/print_wireless_info.c#L126
193    fn nl80211_xbm_to_percent(xbm: f64) -> f64 {
194        const NOISE_FLOOR_DBM: f64 = -90.;
195        const SIGNAL_MAX_DBM: f64 = -20.;
196        let xbm = xbm.clamp(NOISE_FLOOR_DBM, SIGNAL_MAX_DBM);
197        100. - 70. * ((SIGNAL_MAX_DBM - xbm) / (SIGNAL_MAX_DBM - NOISE_FLOOR_DBM))
198    }
199
200    pub async fn new(iface_index: i32) -> Result<Option<Self>> {
201        let mut socket = match AsyncSocket::connect() {
202            Ok(s) => s,
203            Err(_) => return Ok(None),
204        };
205
206        let interfaces = socket
207            .get_interfaces_info()
208            .await
209            .error("Couldn't get nl80211 interfaces")?;
210
211        for iface in interfaces {
212            if let Some(index) = iface.index {
213                if index != iface_index {
214                    continue;
215                }
216
217                let Ok(ap) = socket.get_station_info(index).await else { continue; };
218                let bss = socket
219                    .get_bss_info(index)
220                    .await
221                    .unwrap_or_default()
222                    .into_iter()
223                    .find(|b| b.status == Some(1));
224
225                let signal = ap
226                    .signal
227                    .or(bss.as_ref().and_then(|b| b.signal.map(|s| (s / 100) as i8)));
228
229                let ssid = iface
230                    .ssid
231                    .as_deref()
232                    .map(|v| String::from_utf8_lossy(v).into_owned())
233                    .or_else(|| {
234                        bss.as_ref()
235                            .and_then(|bss| bss.information_elements.as_deref())
236                            .and_then(Self::find_ssid)
237                    });
238
239                return Ok(Some(Self {
240                    ssid,
241                    signal: signal.map(|s| Self::nl80211_xbm_to_percent(s as f64)),
242                    // NL80211_RATE_INFO_BITRATE is specified in units of 100 kbit/s, but is
243                    // used to specify bits/s, so we convert
244                    bitrate: ap.tx_bitrate.map(|br| br as f64 * 1e5),
245                    // MHz -> Hz
246                    frequency: iface.frequency.map(|f| f as f64 * 1e6),
247                }));
248            }
249        }
250        Ok(None)
251    }
252}

254#[derive(Clone, Copy, Debug, SmartDefault, Eq, PartialEq)]
255pub struct RtnlLinkStats {
256    pub rx_bytes: u64,
257    pub tx_bytes: u64,
258}
259
260impl RtnlLinkStats {
261    pub async fn new(
262        socket: &mut NlSocket,
263        if_index: i32,
264    ) -> Result<Option<Self>, Box<dyn StdError + Send + Sync + 'static>> {
265        let ifinfomsg = Ifinfomsg::new(
266            RtAddrFamily::Unspecified,
267            Arphrd::None,
268            0,
269            IffFlags::empty(),
270            IffFlags::empty(),
271            RtBuffer::new(),
272        );
273
274        let mut res = None;
275
276        rtnetlink_recv!(socket, ifinfomsg, Rtm::Getlink, if_index, ifi_index, p: Ifinfomsg => {
277            for rtattr in p.rtattrs.iter() {
278                match rtattr.rta_type {
279                    Ifla::Operstate => {
280                        let state = rtattr.get_payload_as::<u8>()?;
281                        // https://www.kernel.org/doc/html/latest/networking/operstates.html#tlv-ifla-operstate
282                        if state != 6 {
283                            return Ok(None);
284                        }
285                    }
286                    Ifla::Stats64 => {
287                        res = Some(Self::from_rtnl_link_stats64(rtattr.payload().as_ref())?)
288                    }
289                    _ => (),
290                }
291            }
292        });
293
294        Ok(res)
295    }
296
297    // https://www.kernel.org/doc/html/v5.11/networking/statistics.html#c.rtnl_link_stats64 :
298    // struct rtnl_link_stats64 {
299    //   __u64 rx_packets,
300    //   __u64 tx_packets,
301    //   __u64 rx_bytes,
302    //   __u64 tx_bytes,
303    //   -- snip --
304    // }
305    // For the purpose of upload and download speed, we only care about rx_bytes and tx_bytes.
306    pub fn from_rtnl_link_stats64(stats: &[u8]) -> Result<Self> {
307        // We need stats to have at least 32 bytes, which will be read as
308        // 4 blocks of 8 bytes. We only care about the 3rd anf 4th blocks.
309        if stats.len() < 32 {
310            return Err(Error::new(format!(
311                "Bad contents for interface stats: {stats:?}"
312            )));
313        }
314
315        let stats_ptr = stats.as_ptr() as *const u64;
316
317        Ok(Self {
318            rx_bytes: unsafe { stats_ptr.add(2).read_unaligned() },
319            tx_bytes: unsafe { stats_ptr.add(3).read_unaligned() },
320        })
321    }
