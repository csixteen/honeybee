1//! i3bar protocol. For full reference, check the main [documentation].
2//!
3//! The main structures of this module are [`Header`], [`Block`] and [`ClickEvent`]. This is only
4//! relevant if you choose [`i3bar`] as the `output_format`.
5//!
6//! [documentation]: https://i3wm.org/docs/i3bar-protocol.html

15/// The first message in the protocol is a header block.
16#[skip_serializing_none]
17#[derive(Clone, Debug, SmartDefault, Eq, PartialEq, Serialize)]
18pub struct Header {
19    /// The version number (as an integer) of the i3bar protocol you will use.
20    #[default = 1]
21    version: usize,
22
23    /// Specify the signal (as an integer) that i3bar should send to request that you pause your output.
24    /// The default value is SIGSTOP, which will unconditionally stop your process.
25    stop_signal: Option<usize>,
26
27    /// Specify to i3bar the signal (as an integer) to send to continue your processing. The default
28    /// value (if none is specified) is SIGCONT.
29    cont_signal: Option<usize>,
30
31    /// If specified and true i3bar will write an infinite array (same as above) to your stdin.
32    #[default(Some(true))]
33    click_events: Option<bool>,
34}
35
36impl Header {
37    pub fn new() -> Self {
38        Default::default()
39    }
40}

66/// Represents the contents of the status line for a particular module.
67#[skip_serializing_none]
68#[derive(Clone, Debug, SmartDefault, Eq, PartialEq, Serialize)]
69pub struct Block {
70    /// This is the only required key by the protocol. If it's an empty string,
71    /// then the block will be skipped.
72    pub full_text: String,
73    /// If provided, the `short_text` will be used when the status line needs to be shortened
74    /// (because it uses more space than your screen provides).
75    pub short_text: Option<String>,
76    /// The color used to display the contents of `full_text` or `short_text`.
77    #[default(Some(Default::default()))]
78    pub color: Option<Color>,
79    /// Overrides the background color for this particular block.
80    #[default(Some(Color::try_from("#000000").unwrap()))]
81    pub background: Option<Color>,
82    /// Overrides the border color for this particular block.
83    #[default(Some(Color::try_from("#222222").unwrap()))]
84    pub border: Option<Color>,
85    #[default(Some(1))]
86    /// Defines the width (in pixels) of the top border of this block.
87    pub border_top: Option<usize>,
88    /// Defines the width (in pixels) of the right border of this block.
89    #[default(Some(1))]
90    pub border_right: Option<usize>,
91    /// Defines the width (in pixels) of the bottom border of this block.
92    #[default(Some(1))]
93    pub border_bottom: Option<usize>,
94    /// Defines the width (in pixels) of the left border of this block.
95    #[default(Some(1))]
96    pub border_left: Option<usize>,
97    /// The minimum width, in pixels, of this block. If `full_text` takes less
98    /// space than the `min_width`, then the block will be padded to the left
99    /// and/or right, depending on the `alignment`.
100    pub min_width: Option<BlockWidth>,
101    /// Aligns the text on the center, left or right, when the `min_width` is not
102    /// reached.
103    pub align: Alignment,
104    /// Specifies whether the current value is urgent (e.g. no more disk space).
105    pub urgent: Option<bool>,
106    /// Unique name of this block, used to identify it in scripts when processing
107    /// the output.
108    pub name: Option<String>,
109    /// In case there are multiple instances of a block (e.g. multiple disk space
110    /// blocks for multiple mount points).
111    pub instance: Option<String>,
112    /// Indicates whether a separator line should be drawn after this block.
113    #[default(Some(true))]
114    pub separator: Option<bool>,
115    /// Amount of pixels to leave blank after the block. Unless `separator` is disabled,
116    /// a separator line will be drawn in the middle of this gap.
117    #[default(Some(9))]
118    pub separator_block_width: Option<usize>,
119    /// Indicates how the block should be parsed.
120    pub markup: Markup,
121}
122
123impl Block {
124    pub fn new(full_text: String) -> Self {
125        Self {
126            full_text,
127            ..Default::default()
128        }
129    }
130}

139pub fn status_line(rendered_widgets: &[RenderedWidget]) {
140    let v: Vec<_> = rendered_widgets
141        .iter()
142        .filter_map(|r| {
143            if let RenderedWidget::I3Block(b) = r {
144                Some(b.clone())
145            } else {
146                None
147            }
148        })
149        .collect();
150
151    println!("  {},", serde_json::to_string(&v).unwrap());
152}
