7#[derive(Clone, Debug, SmartDefault, PartialEq, Deserialize)]
8#[serde(default)]
9pub struct Config {
10    #[default("default")]
11    device: String,
12    #[default(Format::new().with_default("V: $volume"))]
13    format: Format,
14    #[default(Format::new().with_default("V: muted ($volume)"))]
15    format_muted: Format,
16    #[default("Master")]
17    mixer: String,
18    #[default(0)]
19    mixer_idx: u32,
20}
21
22pub(crate) async fn run(config: Config, bridge: Bridge) -> Result<()> {
23    let mut widget = Widget::new().with_instance(format!(
24        "{}.{}.{}",
25        config.device, config.mixer, config.mixer_idx
26    ));
27    let mut timer = bridge.timer().start();
28
29    loop {
30        let vol = match get_volume(&config)? {
31            Volume::Muted => {
32                widget.set_state(WidgetState::Warning);
33                widget.set_format(config.format_muted.clone());
34                0_f64
35            }
36            Volume::Unmuted(v) => {
37                widget.set_state(WidgetState::Normal);
38                widget.set_format(config.format.clone());
39                v
40            }
41        };
42
43        widget.set_placeholders(map!(
44            "$device" => Value::Text(config.device.clone()),
45            "$volume" => Value::percentage(vol)
46        ));
47
48        bridge.set_widget(widget.clone()).await?;
49
50        loop {
51            tokio::select! {
52                _ = timer.tick() => break,
53            }
54        }
55    }
56}

58#[cfg(not(feature = "pulseaudio"))]
59fn get_volume(config: &Config) -> Result<Volume> {
60    alsa::get_volume(&config.device, &config.mixer, config.mixer_idx)
61}
62
63#[cfg(feature = "pulseaudio")]
64fn get_volume(config: &Config) -> Result<Volume> {
65    pulseaudio::get_volume()
66        .or_else(|_| alsa::get_volume(&config.device, &config.mixer, config.mixer_idx))
67}
