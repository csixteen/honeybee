1//! Gets used, free, available and total amount of bytes on the given mounted filesystem.
2//!
3//! # Configuration
4//!
5//! Key | Description | Values | Default
6//! ----|-------------|--------|--------
7//! `path` | Path to the mounted filesystem | A string representing a path | `"/"`
8//! `format` | A string used to customize the output of this module | See available placeholders below | `"$free"`
9//! `format_below_threshold` | A string to customize the output when state is set to criticalo | See available placeholders below | `"Warning: $percentage_avail"`
10//! `format_not_mounted` | A string to customize the output when the path doesn't exist or is not a mount point. | A string | `""`
11//! `prefix_type` | Prefix used to present byte sizes in human readable format | See options below | `"binary"`
12//! `threshold_type` | Type of `low_threshold` that sets the widget state to critical | See options below | `"percentage_avail"`
13//! `low_threshold` | Value that causes the disk text to be displayed using `color_bad` (widget state critical) | Number | `0`
14//!
15//! Prefix Type | Value
16//! ------------|-------
17//! `binary` | IEC prefixes (KiB, MiB, GiB, TiB) represent multiples of powers of 1024
18//! `decimal` | SI prefixes (K, M, G, T) represent multiples of powers of 1000
19//!
20//! Threshold Type | Value
21//! ---------------|-------
22//! `percentage_free` | Percentage of disk space free
23//! `percentage_avail` | Percentage of disk space available
24//! `bytes_free` | Number of bytes free
25//! `bytes_avail` | Number of bytes available
26//!
27//!  Note that `bytes_free` and `bytes_avail` can be prefixed with "k", "m", "g" or "t". That means that if
28//! you set `prefix_type` to `binary`, `low_threshold` to `2` and `threshold_type` to `gbytes_avail`, then
29//! the disk info will be colored bad.
30//!
31//! If not specified, `threshold_type` is assumed to be `percentage_avail` and `low_threshold` is `0`, which
32//! implies no coloring at all.
33//!
34//! Placeholder | Value
35//! ------------|-------
36//! `$free` | Free disk space
37//! `$percentage_free` | As above, but percentage
38//! `$avail` | Available disk space
39//! `$percentage_avail` | As above, but percentage
40//! `$total` | Total disk space
41//! `$used` | Used disk space
42//! `$percentage_used` | As above, but percentage
43//! `$percentage_used_of_avail` | Percentage of available space being used
44//!
45//! # Example
46//!
47//! ```toml
48//! [[module]]
49//! module = "disk"
50//! format = "/: $avail"
51//! prefix_type = "binary"
52//! threshold_type = "gbytes_avail"
53//! low_threshold = 10
54//! ```
55//!

61#[derive(Clone, Debug, PartialEq, Deserialize, SmartDefault)]
62#[serde(default)]
63pub struct Config {
64    #[default("/")]
65    path: String,
66    #[default(Format::new().with_default("$free"))]
67    format: Format,
68    #[default(Format::new().with_default("Warning: $percentage_avail"))]
69    format_below_threshold: Format,
70    #[default(Format::new().with_default(""))]
71    format_not_mounted: Format,
72    prefix_type: PrefixType,
73    threshold_type: ThresholdType,
74    low_threshold: f64,
75}

77pub(crate) async fn run(config: Config, bridge: Bridge) -> Result<()> {
78    let mut widget = Widget::new().with_instance(config.path.clone());
79    let mut timer = bridge.timer().start();
80    let unit = match config.prefix_type {
81        PrefixType::Binary => Unit::iec_from_str("ti"),
82        _ => Unit::si_from_str("t"),
83    };
84
85    loop {
86        match DiskInfo::new(&config.path) {
87            Ok(disk_info) => {
88                if disk_info.below_threshold(
89                    config.prefix_type,
90                    config.threshold_type,
91                    config.low_threshold,
92                ) {
93                    widget.set_format(config.format_below_threshold.clone());
94                    widget.set_state(WidgetState::Critical);
95                } else {
96                    widget.set_format(config.format.clone());
97                    widget.set_state(WidgetState::Normal);
98                }
99
100                widget.set_placeholders(map!(
101                    "$free" => Value::byte(disk_info.free, unit, 2),
102                    "$used" => Value::byte(disk_info.used, unit, 2),
103                    "$total" => Value::byte(disk_info.total, unit, 2),
104                    "$avail" => Value::byte(disk_info.available, unit, 2),
105                    "$percentage_free" => Value::percentage(disk_info.percentage_free),
106                    "$percentage_used_of_avail" => Value::percentage(disk_info.percentage_used_of_avail),
107                    "$percentage_used" => Value::percentage(disk_info.percentage_used),
108                    "$percentage_avail" => Value::percentage(disk_info.percentage_avail),
109                ));
110            }
111            Err(_) => {
112                widget.set_format(config.format_not_mounted.clone());
113            }
114        }
115
116        bridge.set_widget(widget.clone()).await?;
117
118        loop {
119            tokio::select! {
120                _ = timer.tick() => break,
121            }
122        }
123    }
124}

126#[derive(Clone, Copy, Debug, Eq, PartialEq, Deserialize, SmartDefault)]
127#[serde(rename_all = "lowercase")]
128enum PrefixType {
129    #[default]
130    Binary,
131    Decimal,
132    Custom,
133}
134
135#[derive(Clone, Copy, Debug, Eq, PartialEq, Deserialize, SmartDefault)]
136#[serde(try_from = "String")]
137enum ThresholdType {
138    BytesFree,
139    BytesAvail,
140    PercentageFree,
141    #[default]
142    PercentageAvail,
143    PrefixBytesFree(char),
144    PrefixBytesAvail(char),
145}
146
147impl TryFrom<String> for ThresholdType {
148    type Error = Error;
149
150    fn try_from(value: String) -> std::result::Result<Self, Self::Error> {
151        ThresholdType::try_from(value.as_str())
152    }
153}
154
155impl TryFrom<&str> for ThresholdType {
156    type Error = Error;
157
158    fn try_from(value: &str) -> std::result::Result<Self, Self::Error> {
159        match value {
160            "bytes_free" => Ok(Self::BytesFree),
161            "bytes_avail" => Ok(Self::BytesAvail),
162            "percentage_free" => Ok(Self::PercentageFree),
163            "percentage_avail" => Ok(Self::PercentageAvail),
164            _ => {
165                let unit = value.chars().next().error("Empty string")?;
166                valid_prefix(unit)?;
167                let ttype = value.chars().skip(1).collect::<String>();
168                match ttype.as_str() {
169                    "bytes_free" => Ok(Self::PrefixBytesFree(unit)),
170                    "bytes_avail" => Ok(Self::PrefixBytesAvail(unit)),
171                    _ => Err(Error::new(format!("Invalid threshold type: {value}"))),
172                }
173            }
174        }
175    }
176}
177
178#[inline]
179fn valid_prefix(p: char) -> Result<()> {
180    if !(p == 'k' || p == 'm' || p == 'g' || p == 't') {
181        Err(Error::new("Invalid prefix"))
182    } else {
183        Ok(())
184    }
185}

187#[derive(Clone, Debug, PartialEq, SmartDefault)]
188struct DiskInfo {
189    f_bsize: u64,
190    f_bfree: u64,
191    f_bavail: u64,
192    free: u64,
193    used: u64,
194    total: u64,
195    available: u64,
196    percentage_free: f64,
197    percentage_used_of_avail: f64,
198    percentage_used: f64,
199    percentage_avail: f64,
200}
201
202impl DiskInfo {
203    fn new(path: &str) -> Result<Self> {
204        let res = statvfs::statvfs(path).or_error(|| format!("{path} is not mounted."))?;
205        let f_blocks = res.blocks();
206        let f_bfree = res.blocks_free();
207        let f_bavail = res.blocks_available();
208        let unit_size = if cfg!(any(target_os = "linux", target_os = "netbsd")) {
209            res.fragment_size()
210        } else {
211            res.block_size()
212        };
213
214        Ok(Self {
215            f_bsize: res.block_size(),
216            f_bfree,
217            f_bavail,
218            free: unit_size * f_bfree,
219            used: unit_size * (f_blocks - f_bfree),
220            total: unit_size * f_blocks,
221            available: unit_size * f_bavail,
222            percentage_free: 100_f64 * (f_bfree as f64 / f_blocks as f64),
223            percentage_used_of_avail: 100_f64 * ((f_blocks - f_bavail) as f64) / (f_blocks as f64),
224            percentage_used: 100_f64 * ((f_blocks - f_bfree) as f64) / (f_blocks as f64),
225            percentage_avail: 100_f64 * (f_bavail as f64 / f_blocks as f64),
226        })
227    }
228
229    fn below_threshold(&self, ptype: PrefixType, ttype: ThresholdType, low_threshold: f64) -> bool {
230        match ttype {
231            ThresholdType::PercentageFree => self.percentage_free < low_threshold,
232            ThresholdType::PercentageAvail => self.percentage_avail < low_threshold,
233            ThresholdType::BytesFree => (self.free as f64) < low_threshold,
234            ThresholdType::BytesAvail => (self.available as f64) < low_threshold,
235            ThresholdType::PrefixBytesFree(c) => {
236                prefixed_below_threshold(ptype, self.f_bsize, self.f_bfree, low_threshold, c)
237            }
238            ThresholdType::PrefixBytesAvail(c) => {
239                prefixed_below_threshold(ptype, self.f_bavail, self.f_bfree, low_threshold, c)
240            }
241        }
242    }
243}
244
245#[inline]
246fn prefixed_below_threshold(
247    ptype: PrefixType,
248    value: u64,
249    free: u64,
250    low_threshold: f64,
251    c: char,
252) -> bool {
253    let unit = match ptype {
254        PrefixType::Binary => Unit::iec_from_char(c),
255        _ => Unit::si_from_char(c),
256    };
257    let bytes = unit_to_bytes(low_threshold as u64, unit);
258    value * free < bytes
259}
