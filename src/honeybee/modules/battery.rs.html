1//! Gets the status (charging, discharging, unknown, full), percentage, remaining time and
2//! power consumption (in Watts) of the given battery and when it's estimated to be empty.
3//!
4//! # Configuration
5//!
6//! Key | Description | Values | Default
7//! ----|-------------|--------|--------
8//! `number` | Battery index as reported in `/sys` | An integer | 0
9//! `path` | Set this property if your battery is represented in a non-standard path in `/sys`. The first occurrence of `%d` gets replaced with `number`, but you can also hardcode the battery index in the path | String representing a path | `"/sys/class/power_supply/BAT%d/uevent"`
10//! `format` | A string used to customize the output of this module | See available placeholders below | `"$status $percentage $remaining"`
11//! `format_down` | A string to customize the output when there are no metrics | A string of UTF-8 characters | `"No battery"`
12//! `status_chr` | Custom string to represent `charging` state. | String with any UTF-8 symbols | `"CHR"`
13//! `status_bat` | same as above, but for `discharging`. | same as above | `"BAT"`
14//! `status_unk` | same as above, but for `unknown` | same as above | `"UNK"`
15//! `status_full` | same as above, but for `full` | same as above | `"FULL"`
16//! `low_threshold` | Causes the widget state to change to [`WidgetState::Critical`] | Integer | `10`
17//! `threshold_type` | An integer representing either time or percentage. If you define `percentage` and your battery percentage goes below `low_threshold`, the widget state will change accordingly. | `time`, `percentage` | `percentage`
18//!
19//! Placeholder | Value
20//! ------------|-------
21//! `$status` | Status (charging, discharging, unknown, full) of the battery. Will be replaced by one of the `status_*` properties defined above.
22//! `$percentage` | Battery left percentage
23//! `$remaining` | Remaining time
24//! `$emptytime` | Time when it's estimated to be empty
25//! `$consumption` | Power consumption in Watts
26//!
27//! # Example
28//! ```toml
29//! [[module]]
30//! module = "battery"
31//! number = 0
32//! format = "$status $percentage $remaining"
33//! format_down = "No battery"
34//! status_chr = "⚇ CHR"
35//! status_bat = "⚡ BAT"
36//! status_full = "☻ FULL"
37//! low_threshold = 20
38//! threshold_type = "percentage"
39//! ```
40//!

74pub(crate) async fn run(config: Config, bridge: Bridge) -> Result<()> {
75    let mut widget = Widget::new();
76    let mut timer = bridge.timer().start();
77    let path = config.path.replace("%d", &config.number.to_string());
78
79    loop {
80        let reader = buffered_reader(&path).await?;
81        let psi = PowerSupplyInfo::new(reader, config.last_full_capacity).await?;
82        let bi = BatteryInfo::from(&psi);
83
84        if bi.full <= 0 && psi.remaining <= 0 && psi.percentage_remaining <= 0 {
85            widget.set_format(config.format_down.clone());
86        } else {
87            widget.set_format(config.format.clone());
88            widget.set_state(if is_battery_low(&config, &bi) {
89                WidgetState::Critical
90            } else {
91                WidgetState::Normal
92            });
93
94            widget.set_placeholders(map!(
95                "$status" => Value::Text(battery_status(&config, &bi).to_owned()),
96                "$percentage" => Value::percentage(bi.percentage_remaining),
97                "$remaining" => Value::Text(bi.time_remaining),
98                "$emptytime" => Value::Text(bi.empty_time),
99                "$consumption" => Value::Text(bi.consumption)
100            ));
101        }
102
103        bridge.set_widget(widget.clone()).await?;
104
105        loop {
106            tokio::select! {
107                _ = timer.tick() => break,
108            }
109        }
110    }
111}

113fn is_battery_low(config: &Config, bi: &BatteryInfo) -> bool {
114    if config.low_threshold <= 0 || bi.status != ChargingStatus::Discharging {
115        return false;
116    }
117
118    match config.threshold_type {
119        ThresholdType::Percentage if bi.percentage_remaining >= 0_f64 => {
120            (bi.percentage_remaining as i64) < config.low_threshold
121        }
122        ThresholdType::Time if bi.seconds_remaining >= 0_f64 => {
123            (bi.seconds_remaining as i64) < 60 * config.low_threshold
124        }
125        _ => false,
126    }
127}

129fn battery_status<'a>(config: &'a Config, info: &BatteryInfo) -> &'a str {
130    match info.status {
131        ChargingStatus::Unknown => &config.status_unk,
132        ChargingStatus::Charging => &config.status_chr,
133        ChargingStatus::Discharging => &config.status_bat,
134        ChargingStatus::Full => &config.status_full,
135    }
136}

146#[derive(Clone, Debug, SmartDefault)]
147struct BatteryInfo {
148    full: i64,
149    status: ChargingStatus,
150    percentage_remaining: f64,
151    seconds_remaining: f64,
152    time_remaining: String,
153    empty_time: String,
154    consumption: String,
155}
156
157impl From<&PowerSupplyInfo> for BatteryInfo {
158    fn from(psi: &PowerSupplyInfo) -> Self {
159        let mut bi = BatteryInfo::default();
160        bi.status = psi.status;
161
162        bi.full = if psi.full_design <= 0 || (psi.last_full_capacity && psi.full_last > 0) {
163            psi.full_last
164        } else {
165            psi.full_design
166        };
167
168        bi.percentage_remaining = if psi.percentage_remaining < 0 {
169            let p = (psi.remaining as f64 / bi.full as f64) * 100_f64;
170            if psi.last_full_capacity && p > 100_f64 {
171                100_f64
172            } else {
173                p
174            }
175        } else {
176            psi.percentage_remaining as f64
177        };
178
179        bi.seconds_remaining = if psi.seconds_remaining < 0
180            && psi.present_rate > 0
181            && bi.status != ChargingStatus::Full
182        {
183            match bi.status {
184                ChargingStatus::Charging => {
185                    (3600 * (bi.full - psi.remaining)) as f64 / psi.present_rate as f64
186                }
187                ChargingStatus::Discharging => {
188                    (3600 * psi.remaining) as f64 / psi.present_rate as f64
189                }
190                _ => 0_f64,
191            }
192        } else {
193            psi.seconds_remaining as f64
194        };
195
196        bi.empty_time = {
197            let et = Local::now() + Duration::seconds(bi.seconds_remaining as i64);
198            format!("{}", et.format("%H:%M:%S"))
199        };
200
201        if bi.seconds_remaining >= 0_f64 {
202            let hours = (bi.seconds_remaining / 3600_f64) as i64;
203            let mut seconds = bi.seconds_remaining as i64 - (hours * 3600);
204            let minutes = seconds / 60;
205            seconds -= minutes * 60;
206            bi.time_remaining = format!("{:02}:{:02}:{:02}", hours, minutes, seconds);
207        }
208
209        bi.consumption = format!("{:1.2}W", psi.present_rate as f64 / 1.0E+6_f64);
210
211        bi
212    }
213}

215#[derive(Debug, Clone, SmartDefault)]
216struct PowerSupplyInfo {
217    #[default(-1)]
218    full_last: i64,
219    #[default(-1)]
220    full_design: i64,
221    #[default(-1)]
222    remaining: i64,
223    #[default(-1)]
224    present_rate: i64,
225    #[default(-1)]
226    seconds_remaining: i32,
227    #[default(-1)]
228    voltage: i64,
229    #[default(-1)]
230    percentage_remaining: i32,
231    status: ChargingStatus,
232    watt_as_unit: bool,
233    last_full_capacity: bool,
234}
235
236impl PowerSupplyInfo {
237    /// Reference: https://www.kernel.org/doc/html/latest/power/power_supply_class.html
238    async fn new<T>(mut reader: BufReader<T>, last_full_capacity: bool) -> Result<Self>
239    where
240        T: AsyncRead + Unpin,
241    {
242        let mut info = PowerSupplyInfo {
243            last_full_capacity,
244            ..Default::default()
245        };
246
247        let mut line = String::new();
248
249        // Ok(0) means that EOF was reached
250        while reader
251            .read_line(&mut line)
252            .await
253            .error("Error reading line.")?
254            != 0
255        {
256            if line.is_empty() {
257                continue;
258            }
259
260            let mut s = line.trim().split('=');
261            let field = s.next().error("No field")?;
262            let value = s.next().error("No value")?;
263
264            match field {
265                // Momentary energy value
266                "POWER_SUPPLY_ENERGY_NOW" => {
267                    info.watt_as_unit = true;
268                    info.remaining = from_str!(i64, value, field);
269                    info.percentage_remaining = -1;
270                }
271                // Momentary charge value
272                "POWER_SUPPLY_CHARGE_NOW" => {
273                    info.watt_as_unit = false;
274                    info.remaining = from_str!(i64, value, field);
275                    info.percentage_remaining = -1;
276                }
277                // Attribute represents capacity in percents (from 0 to 100)
278                "POWER_SUPPLY_CAPACITY" => {
279                    if info.remaining == -1 {
280                        info.percentage_remaining = from_str!(i32, value, field)
281                    }
282                }
283                "POWER_SUPPLY_CURRENT_NOW" => info.present_rate = from_str!(i64, value, field),
284                // Momentary power supply voltage value.
285                "POWER_SUPPLY_VOLTAGE_NOW" => info.voltage = from_str!(i64, value, field),
286                // Seconds left for battery to be considered empty (i.e. while
287                // battery powers a load)
288                "POWER_SUPPLY_TIME_TO_EMPTY_NOW" => {
289                    info.seconds_remaining = from_str!(i32, value, field) * 60
290                }
291                "POWER_SUPPLY_POWER_NOW" => info.present_rate = from_str!(i64, value, field),
292                // See [`ChargingStatus`]
293                "POWER_SUPPLY_STATUS" => info.status = ChargingStatus::from(value),
294                // Design charge and energy values, respectively, when battery
295                // considered full.
296                "POWER_SUPPLY_CHARGE_FULL_DESIGN" | "POWER_SUPPLY_ENERGY_FULL_DESIGN" => {
297                    info.full_design = from_str!(i64, value, field)
298                }
299                // Last remembered value of energy and charge, respectively, when battery
300                // became full. These attributes represent real thresholds, not design
301                // values (i.e. depend on conditions such temperature or age).
302                "POWER_SUPPLY_ENERGY_FULL" | "POWER_SUPPLY_CHARGE_FULL" => {
303                    info.full_last = from_str!(i64, value, field)
304                }
305                _ => (),
306            }
307
308            line.clear();
309        }
310
311        Ok(info)
312    }
313}

315/// Represents the operating status. Corresponds to
316/// `POWER_SUPPLY_STATUS_*`, as defined in [`power_supply.h`].
317///
318/// [`power_supply.h`]: https://github.com/torvalds/linux/blob/master/include/linux/power_supply.h
319#[derive(Clone, Copy, Debug, SmartDefault, Eq, PartialEq)]
320pub enum ChargingStatus {
321    Unknown,
322    Charging,
323    #[default]
324    Discharging,
325    Full,
326}
327
328impl From<&str> for ChargingStatus {
329    fn from(status: &str) -> Self {
330        match status {
331            "Charging" => Self::Charging,
332            "Discharging" | "Not charging" => Self::Discharging,
333            "Full" => Self::Full,
334            _ => Self::Unknown,
335        }
336    }
337}
